Lemma overflow_check_OK_exist_int: forall v v',
  overflowCheck v (OK v') ->
    v' = Int v.
Proof.
  intros.
  remember (OK v') as x.
  inversion H; smack.
Qed.

    specialize (overflow_check_OK_exist_int _ _ H);
    let HZ := fresh "HZ" in intros HZ
  end.
